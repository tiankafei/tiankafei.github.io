# centos7安装jira

# 设置可执行权限

chmod u+x atlassian-jira-software-9.3.1-x64.bin

# 执行安装

./atlassian-jira-software-9.3.1-x64.bin

# jira 修改setenv.sh文件

cd /opt/atlassian/jira/bin
vim setenv.sh

# 一直下拉，在末位加入如下内容（请注意其中路径，如果未完全按照文档，请自行修改）
export JAVA_OPTS="-javaagent:/opt/software/atlassian-agent.jar ${JAVA_OPTS}"

# 复制mysql启动到jira安装目录

cp /opt/software/mysql-connector-java-8.0.30.jar /opt/atlassian/jira/atlassian-jira/WEB-INF/lib/

# jira启动、停止

service jira stop
service jira start

# jira 许可证

java -jar atlassian-agent.jar -p jira -m 798971170@qq.com -n tiankafei -o http://192.168.20.101:9000 -s B209-N8LE-U169-FEDW

-p jira 产品名称
-m 邮箱
-n 名称
-o 访问的url
-s 服务器ID

# confluence

service confluence stop
service confluence start

# confluence破解

cd /opt/atlassian
mkdir confluence
tar -zxvf /opt/software/atlassian-confluence-7.16.5.tar.gz -C /opt/atlassian/confluence

# 编辑confluence初始化配置
vim /opt/atlassian/confluence/confluence/WEB-INF/classes/confluence-init.properties
# 添加一行配置
confluence.home=/opt/atlassian/confluence/data

# 备份atlassian-extras-decoder-v2-3.4.6.jar
cp /opt/atlassian/confluence/confluence/WEB-INF/lib/atlassian-extras-decoder-v2-3.4.1.jar /opt/

atlassian-extras-decoder-v2-3.4.6.jar	改名	atlassian-extras-2.4.jar
# 执行 java -jar confluence_keygen.jar
# mysql驱动拷贝至指定目录
cp /opt/software/mysql-connector-java-8.0.30.jar /opt/atlassian/confluence/confluence/WEB-INF/lib/

# 更新mysql配置

vim /etc/my.cnf

lower_case_table_names=1
default-storage-engine=INNODB
character_set_server=utf8mb4
collation-server=utf8mb4_bin
max_allowed_packet=256M
innodb_log_file_size=2GB
innodb_default_row_format=DYNAMIC
transaction-isolation=READ-COMMITTED
sql_mode = NO_AUTO_VALUE_ON_ZERO
log_bin_trust_function_creators = 1

systemctl restart mysqld.service

# confluence开机启动

vim /lib/systemd/system/confluence.service

[Unit]
Description=confluence
After=network.target

[Service]
Type=forking
ExecStart=/opt/atlassian/confluence/bin/start-confluence.sh
ExecStop=/opt/atlassian/confluence/bin/stop-confluence.sh
PrivateTmp=true

[Install]
WantedBy=multi-user.target

# 重新加载配置
systemctl daemon-reload
# 设置开机启动
systemctl enable confluence.service
# 启动
systemctl start confluence.service
# 停止
systemctl stop confluence.service
# 查看状态
systemctl status confluence.service
